function Q_函数:取天降灵猴信息(任务id,玩家id,序号)--没问题啊
	local 战斗单位={}
	local 等级=取队伍平均等级(玩家数据[玩家id].队伍,玩家id)
	for n=1,10 do
		战斗单位[n]={
		名称="下凡的灵猴"
		,模型="超级六耳猕猴"
		,伤害=(等级-106)*8+1200
		,气血=(等级-106)*80+7000
		,法伤=等级*5
		,速度=等级*2
		,防御=等级*6
		,法防=等级
		,躲闪=等级*2
		,魔法=200
		,等级=等级
		,技能={}
		,主动技能=取随机法术(3)
		}
	end
	战斗单位[1].模型=任务数据[任务id].模型

战斗单位 = self:置怪物难度配置(战斗单位)
	return 战斗单位
end
