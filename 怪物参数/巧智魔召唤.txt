function Q_函数:巧智魔召唤(等级)
	等级=等级-5
	 if 等级<20 then 等级=20 end
	 local sx = self:取属性(等级) --类型：固伤 辅助 物理 法系
	 local 召唤单位={
		  名称=" 巧智魔 "
		  ,模型="蜃气妖"
		  ,等级=等级
		  ,气血 = qz(sx.属性.气血 * 0.8)
		  ,伤害 = qz(sx.属性.伤害 * 0.6)
		  ,法伤 = qz(sx.属性.法伤 * 0.6)
		  ,速度 = qz(sx.属性.速度 * 0.6)
		  ,主动技能=sx.技能组
	 }
战斗单位 = self:置怪物难度配置(战斗单位)
	 return 召唤单位
end
